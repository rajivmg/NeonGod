#version 330

uniform mat4 MVP;
uniform mat4 M;

in  vec3 vertex_position;
in 	vec3 vertex_normal;
in  vec2 vertex_uv; 

out vec3 fragVert;
out vec2 UV;
out vec3 fragNormal;

void main () {

  fragNormal = vertex_normal;
  UV = vertex_uv;
  fragVert = vertex_position;

  gl_Position = MVP * vec4(vertex_position, 1.0);
}
